PCS Score

Computation

Mean subjective scale responses for item 4 (Taste) and item 10 (Post-session suggestion) are each generated from the combination of two responses which require processing prior to generating an overall subjective score. The subjective score for item 4 is the mean of the responses to the (a)‘sour’ and (b)‘sweet’ sub-scales for the taste suggestion. The subjective score for item 10 is the geometric mean (the square root of the product) of the responses to the (a) ‘urge’ and (b) ‘amnesia’ sub-scales for the post-session suggestion.

  bounds <- bayestestR::hdi(ig_scores[[var]], ci = 0.975)
  cis <- rbind(cis, data.frame(variable = var, CI_low = bounds$CI_low, CI_high = bounds$CI_high))

  # Make plot
  ps[[var]] <- ggplot(ig_scores, aes(x = !!as.name(var))) +
    geom_histogram(binwidth = 0.015) +
    geom_vline(xintercept = bounds$CI_low, color = "red") +
    geom_vline(xintercept = bounds$CI_high, color = "red") +
    coord_cartesian(xlim = c(0, 1))  +
    ggtitle(paste("97.5% HDI for", var))   +
    theme_minimal()
  
  illu <- str_split_fixed(var, "_", 4)[,2]
  flagged_ppts <- ig_scores$Participant[(ig_scores[[var]] > bounds$CI_high) | (ig_scores[[var]] < bounds$CI_low) |  is.na(ig_scores[[var]])]
  # print(length(flagged_ppts))
  outliers[[illu]] <- unique(c(outliers[[illu]], flagged_ppts))
}

Manipulation Check

Significant differences between the Congruent and Mild and the Mild and Strong groups are taken as evidence that illusions functioned as intended.

ttests <- data.frame()
for (outcome in c("ErrorRate", "IES")) {
    for (ill_type in c("Ebbinghaus", "MullerLyer", "VerticalHorizontal")) {
        for (cond in c("Congruent", "Mild")) {
            x1 <- ig_scores[[paste0(outcome, "_", ill_type, "_", cond, "_General")]]
            if (cond == "Congruent") {
                x2 <- ig_scores[[paste0(outcome, "_", ill_type, "_", "Mild_General")]]
            } else {
                x2 <- ig_scores[[paste0(outcome, "_", ill_type, "_", "Strong_General")]]
            }

            complete <- complete.cases(x1, x2)
            x1 <- x1[complete]
            x2 <- x2[complete]

            if (length(x1) < 2) {
                message(paste("Skipping:", outcome, ill_type, cond, "- not enough observations"))
                next
            }

            ttests <- BayesFactor::ttestBF(x = x1, y = x2, paired = TRUE) |>
                parameters::parameters() |>
                mutate(
                    Outcome = outcome,
                    Illusion_Type = ill_type,
                    Condition_Illusion = cond
                ) |>
                rbind(ttests)
        }
    }
}

make_cor <- function(data) {
  features <- merge(ig_scores, data[c("Participant", "PCS_Score", "PID5_Psychoticism")])
  
  rez <- correlation::correlation(
    dplyr::select(features, -PCS_Score, -PID5_Psychoticism),
    dplyr::select(features, PCS_Score, PID5_Psychoticism),
    bayesian = TRUE
  )
  
  table <- rez |>
    dplyr::arrange(desc(BF)) |> 
    format_table() |>
    dplyr::mutate(BF = as.numeric(BF),
                  Parameter2 = str_remove_all(Parameter2, "_Score|PID5_")) |> 
    dplyr::select(-Prior, -Method, -n_Obs, -`% in ROPE`) |>
    gt::gt() |>
    gt::data_color(
      columns = BF,
      fn = function(bf) dplyr::case_when(
        bf >= 10   ~ "green",
        bf >= 3    ~ "lightgreen",
        bf <= 1/10 ~ "red",
        bf <= 1/3  ~ "#ffabab",
        .default   = "white"
      )
    )
  
  assign("features", features, envir = .GlobalEnv)
  assign("rez", rez, envir = .GlobalEnv)
  
  table
}
